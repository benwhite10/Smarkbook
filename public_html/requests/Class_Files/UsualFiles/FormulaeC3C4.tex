\subsection*{Trigonometric identities}
\begin{flalign*}
\sin(A\pm B)&=\sin A\cos B \pm \cos A\sin B&\\
\cos(A\pm B)&=\cos A\cos b \mp \sin A\sin B&\\
\tan(A\pm B)&=\dfrac{\tan A\pm\tan B}{1\mp\tan A \tan B}\qquad \left(A\pm B\neq \left(k+\frac{1}{2}\right)\pi\right)&\\
\end{flalign*}
\subsection*{Differentiation}
\begin{flalign*}
\mathbf{\fx}\qquad&\mathbf{\fdx}&\\
\tan kx\qquad& k\sec^2kx&\\
\sec x\qquad&\sec x\tan x&\\
\cot x\qquad&-\csc^2 x&\\
\csc x\qquad&-\csc x\cot x&\\
\dfrac{\fx}{\gx}\qquad&\dfrac{\fdx\gx-\fx\gdx}{\left(\gx\right)^2}&
\end{flalign*}
\subsection*{Integration}

\cot x\qquad&\ln|\sin x|&\\
\csc x\qquad&-\ln|\csc x+\cot x|=\ln\Big|\tan\left(\frac{1}{2}x\right)\Big|&\\
\sec x\qquad& \ln|\sec x + \tan x|=\ln\Big|\tan\left(\frac{1}{2}x+\frac{1}{4}\pi\right)\Big|
\end{flalign*}
